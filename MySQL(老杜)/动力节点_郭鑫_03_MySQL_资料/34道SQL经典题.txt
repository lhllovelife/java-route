select 
       sc.sno,s.sname,count(*) as studentNum

    s

     sc.sno = s.sno

t2

       sc.sno,avg(sc.scgrade) as avgscgrade

       sc

       t1.sname,t2.avgscgrade

       (select 
               sc.sno,s.sname,count(*) as studentNum

             sc.sno = s.sno
